## 二、DNA 完整性稽核

### 2.1 定期扫描脚本

```typescript
// scripts/audit-dna-integrity.ts
// 定时扫描全量表，输出畸形 DNA 清单

async function auditDNAIntegrity(): Promise<AuditReport> {
  const issues: DNAIssue[] = [];

  // 1. 扫描 state_spines 表
  const spines = db.all('SELECT * FROM state_spines');
  for (const row of spines) {
    if (!DNA.validate(row.dna_branch)) {
      issues.push({ type: 'invalid_dna', globalUID: row.dna_id, detail: 'spine branch' });
    }
    if (row.dimension_id < 1 || row.dimension_id > 32) {
      issues.push({ type: 'dimension_oob', globalUID: row.dna_id, detail: String(row.dimension_id) });
    }
  }

  // 2. 扫描 atom_address_timeline 表
  const addrs = db.all('SELECT * FROM atom_address_timeline');
  for (const row of addrs) {
    if (!DNA.validateTimeline(row.global_uid, row.crc_checksum)) {
      issues.push({ type: 'crc_mismatch', globalUID: row.global_uid });
    }
    if (row.state_flag === 'F') {
      issues.push({ type: 'confirmed_corrupted', globalUID: row.global_uid });
    }
  }

  // 3. 检查孤立记录（语义链有但寻址链无，或反之）
  const orphans = db.all(`
    SELECT s.dna_id FROM state_spines s
    LEFT JOIN atom_address_timeline a ON s.dna_id = a.global_uid
    WHERE a.global_uid IS NULL
    UNION
    SELECT a.global_uid FROM atom_address_timeline a
    LEFT JOIN state_spines s ON a.global_uid = s.dna_id
    WHERE s.dna_id IS NULL
  `);
  for (const row of orphans) {
    issues.push({ type: 'orphan_record', globalUID: row.dna_id || row.global_uid });
  }

  return { total: spines.length / 32 + addrs.length, issues, timestamp: Date.now() };
}
```

### 2.3 稽核频率

| 稽核项 | 频率 | 触发 |

```bash
# CI 检查：是否有模块绕过全局编解码器
grep -rn "state_spines.*INSERT\|atom_address.*INSERT" src/ \
  --include="*.ts" \
  --exclude="src/m2/*" \
  --exclude="src/transcoder/*"
```
